Statistical Analysis 2: Hypothesis Test — Has Disease Rate Changed Over Time?¶

Hypothesis:

  • H₀ (Null): There is no significant difference in average disease rates before and after 2015.
  • H₁ (Alternative): There is a significant difference in average disease rates before and after 2015.

We use a Welch's two-sample t-test (does not assume equal variance).

In [29]:
# Dynamically find the midpoint year to split the data
years = sorted(df_new["YearStart"].dropna().unique())
split_year = years[len(years) // 2]
print(f"Years in dataset: {years}")
print(f"Splitting at year: {split_year}\n")

early = df_new[df_new["YearStart"] < split_year]["DataValue"].dropna()
late  = df_new[df_new["YearStart"] >= split_year]["DataValue"].dropna()

# Welch's t-test
t_stat, p_val = stats.ttest_ind(early, late, equal_var=False)

print("=== Welch's Two-Sample T-Test ===")
print(f"  Early period (< {split_year})  — Mean: {early.mean():.4f}, N: {len(early):,}")
print(f"  Late period  (≥ {split_year}) — Mean: {late.mean():.4f}, N: {len(late):,}")
print(f"  T-statistic: {t_stat:.4f}")
print(f"  P-value:     {p_val:.6f}")
print(f"  Conclusion:  {'Reject H₀ — significant difference found (p < 0.05)' if p_val < 0.05 else 'Fail to reject H₀ — no significant difference (p ≥ 0.05)'}")

# Visualize the two distributions
n_early = min(3000, len(early))
n_late  = min(3000, len(late))

plt.figure(figsize=(10, 5))
plt.hist(early.sample(n_early, random_state=1), bins=50, alpha=0.5, density=True, color='steelblue', label=f'Early (< {split_year})')
plt.hist(late.sample(n_late, random_state=1),   bins=50, alpha=0.5, density=True, color='coral',     label=f'Late (≥ {split_year})')
plt.axvline(early.mean(), color='steelblue', linestyle='--', linewidth=2, label=f'Early Mean: {early.mean():.2f}')
plt.axvline(late.mean(),  color='coral',     linestyle='--', linewidth=2, label=f'Late Mean: {late.mean():.2f}')
plt.title(f"Disease Rate Distribution: Early vs Late Period (split at {split_year})", fontsize=13)

from scipy.stats import linregress

year_avg = df_new.groupby("YearStart")["DataValue"].mean().reset_index()
